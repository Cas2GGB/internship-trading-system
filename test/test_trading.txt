# ===============================================================
# 混合账户与资金测试用例: 挂单(Maker), 吃单(Taker), 穿透成交, 撤单(Cancel)
# 格式: ADD <StockID> <OrderID> <Price> <Qty> <Side 1=Buy/2=Sel> <ClientID>
#        CANCEL <StockID> <OrderID>
# ===============================================================

# 0. 初始化账户资金/持仓 (给空头卖家发放股票)
# 买家群体 ClientID: 10, 11
# 卖家群体 ClientID: 20, 21
GIVE_POS 20 1 500
GIVE_POS 21 1 500

echo "------ 初始账户状态 ------"
ACCOUNT 10
ACCOUNT 20

# 1. 建立初始盘口 (Maker) 
# 买一: 100元 50股 (ID: 100) -> 账户 10
# 买二:  99元 50股 (ID: 101) -> 账户 11
ADD 1 100 100 50 1 10
ADD 1 101 99 50 1 11
# 卖一: 101元 50股 (ID: 102) -> 账户 20
# 卖二: 102元 50股 (ID: 103) -> 账户 21
ADD 1 102 101 50 2 20
ADD 1 103 102 50 2 21

echo "------ 建仓后账户状态(应出现资金/持仓冻结) ------"
ACCOUNT 10
ACCOUNT 20

# 2. 刚好完全成交测试
# 买单 (ID: 200, 价格101, 数量50), 账户 10 发起, 将完全吃掉卖一(ID: 102)
# 成交价应该是被动盘的价格 101
ADD 1 200 101 50 1 10

# 3. 故意下一个资金不足/持仓不足被拒的违规单
# 账户 99 (没给他发股票) 尝试卖出
ADD 1 999 100 1000 2 99

# 4. 穿透扫单测试与部分成交
# 卖单 (ID: 202, 价格90较低, 数量70), 账户 20 砸盘
# 会吃掉所有的买一(50股 @ 100) 和买二的(20股 @ 99)
ADD 1 202 90 70 2 20

echo "------ 剧烈交易后账户状态 ------"
ACCOUNT 10
ACCOUNT 11
ACCOUNT 20

# 5. 撤单功能测试：撤单解冻验证
# 尝试撤销已被部分吃掉的买二 101 (原50, 被吃了20, 剩30)
# 预期：撤销后，对应 30 股的冻结买入资金(30*99=2970)原路返回可用余额
CANCEL 1 101

# 尝试撤销已被只部分成交的卖单 (103, 原50, 因为没人买它，全剩着)
# 预期：全额退回冻结持仓
CANCEL 1 103

echo "------ 撤单解冻结后最终账户状态 ------"
ACCOUNT 11
ACCOUNT 21

EXIT
